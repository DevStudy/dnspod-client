Dnspod-client

Dnspod-client is a client of DNSPod. It was originally designed for use with node.js.

Quick Examples

var Dnspod = require('dnspod-client');

var client = new Dnspod({
    'login_email': 'test@test.com',
    'login_password': 'test'
});

client.on('domainList',function(err, data){
    if (err) {
        throw err;
    } else {
        // done();
    }
}).domainList({length:5});

client.on('getHostIp', function(err, message){
    if (err) {
        throw err;
    } else {
        console.log('get IP address: ' + message);
        // done();
    }
}).getHostIp();

Download

You can install using Node Package Manager (npm):

npm install dnspod-client
